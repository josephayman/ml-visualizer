# To run the frontend use these commands
- cd frontend
- npm install
- npm start

# To run the backend use these commands (windows)
- cd backend
- py -3 -m venv venv
- venv\Scripts\activate
- pip install Flask
- python server.py

# To run the backend use these commands (mac)
- cd backend
- python3 -m venv venv
- source venv/bin/activate
- pip3 install Flask
- python3 server.py
